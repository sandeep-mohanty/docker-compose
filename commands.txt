# Normal Mode
docker-compose -f base.yml -f database.yml -f redis.yml -f rabbit.yml -f mongo.yml up
docker-compose -f base.yml -f database.yml -f redis.yml -f rabbit.yml -f mongo.yml down

# De-attached Mode
docker-compose -f base.yml -f database.yml -f redis.yml -f rabbit.yml -f mongo.yml up -d
docker-compose -f base.yml -f database.yml -f redis.yml -f rabbit.yml -f mongo.yml down -d
